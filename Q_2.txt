
#include <stdio.h>
#include <string.h>

char *capitalLetters(char* str, int len){
    char *answer = malloc(len);
    int index = 0;
    int i=0;
    while (str[i] != '\0'){
        if (str[i] >= 'A' && str[i] <= 'Z'){
            answer[index] = str[i];
            index++;
        }
        i++;
    }
    answer[index] = '\0';
    return answer;
    free(answer);
}

int main()
{
    char cap[20] = "#HeLLo7";
    printf("%s", capitalLetters(cap, 20));
    
}
